109  template<class F1, class F2, class F3>
110  BDD compute_gb(const Cudd& manager, F1& system_post, F2& radius_post, F3&& avoid, size_t& no_trans) {
111  /* number of cells */
112  abs_type N=m_pre.size();
113  /* number of inputs */
114  abs_type M=m_input.size();
115  /* state space dimension */
116  int dim=m_pre.get_dim();
117  /* for display purpose */
118  abs_type counter=0;
119  /* variables for managing the post */
120  std::vector<abs_type> lb(dim); /* lower-left corner */
121  std::vector<abs_type> ub(dim); /* upper-right corner */
122  /* radius of hyper interval containing the attainable set */
123  state_type eta;
124  state_type r;
125  /* state and input variables */
126  state_type x;
127  input_type u;
128  /* for out of bounds check */
129  state_type lower_left;
130  state_type upper_right;
131  /* copy data from m_state_alphabet */
132  for(int i=0; i<dim; i++) {
133  eta[i]=m_pre.get_eta()[i];
134  lower_left[i]=m_pre.get_lower_left()[i];
135  upper_right[i]=m_pre.get_upper_right()[i];
136  }
137  /* the BDD to encode the transition function */
138  BDD tf = manager.bddZero();
139  /* is post of (i,j) out of domain ? */
140  bool out_of_domain;
141  /* loop over all cells */
142  for(abs_type i=0; i<N; i++) {
143  BDD bdd_i = m_pre.id_to_bdd(i);
144  /* is i an element of the avoid symbols ? */
145  if(avoid(i)) {
146  continue;
147  }
148  /* loop over all inputs */
149  for(abs_type j=0; j<M; j++) {
150  BDD bdd_j = m_input.id_to_bdd(j);
151  /* get center x of cell */
152  m_pre.itox(i,x);
153  /* cell radius (including measurement errors) */
154  for(int k=0; k<dim; k++)
155  r[k]=eta[k]/2.0+m_z[k];
156  /* current input */
157  m_input.itox(j,u);
158  /* integrate system and radius growth bound */
159  /* the result is stored in x and r */
160  radius_post(r,x,u);
161  system_post(x,u);
162  /* determine the cells which intersect with the attainable set:
163  * discrete hyper interval of cell indices
164  * [lb[0]; ub[0]] x .... x [lb[dim-1]; ub[dim-1]]
165  * covers attainable set */
166  for(int k=0; k<dim; k++) {
167  /* check for out of bounds */
168  double left = x[k]-r[k]-m_z[k];
169  double right = x[k]+r[k]+m_z[k];
170  if(left <= lower_left[k]-eta[k]/2.0 || right >= upper_right[k]+eta[k]/2.0) {
171  out_of_domain=true;
172  break;
173  }
174  /* integer coordinate of lower left corner of post */
175  lb[k] = static_cast<abs_type>((left-lower_left[k]+eta[k]/2.0)/eta[k]);
176  /* integer coordinate of upper right corner of post */
177  ub[k] = static_cast<abs_type>((right-lower_left[k]+eta[k]/2.0)/eta[k]);
178  }
179  if(out_of_domain) {
180  out_of_domain=false;
181  continue;
182  }
183  /* compute BDD of post */
184  BDD bdd_k = m_post.interval_to_bdd(manager,lb,ub);
185  /* add to transition function */
186  tf = tf | (bdd_i & bdd_j & bdd_k);
187  }
188  /* print progress */
189  progress(i,N,counter);
190  }
191 
192  /* count number of transitions */
193  size_t nvars = m_pre.get_no_bdd_vars() +
194  m_input.get_no_bdd_vars() +
195  m_post.get_no_bdd_vars();
196  no_trans=tf.CountMinterm(nvars);
197  return tf;
198  }
