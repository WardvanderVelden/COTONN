96  SymbolicSet(const SymbolicSet& set1, const SymbolicSet& set2) {
97  m_dim = set1.m_dim + set2.m_dim;

100  m_no_grid_points.reset(new abs_type[m_dim]);

105  m_no_grid_points[i] = set1.m_no_grid_points[i];
106  m_bdd_interval.push_back(set1.m_bdd_interval[i]);

108  for(int j=set1.m_dim, i=0; i<set2.m_dim; i++) {
109  m_eta[j+i] = set2.m_eta[i];
110  m_first[j+i] = set2.m_first[i];
111  m_no_grid_points[j+i] = set2.m_no_grid_points[i];
112  m_bdd_interval.push_back(set2.m_bdd_interval[i]);

135  BDD id_to_bdd(abs_type id) const {
136  abs_type num;
137  int k=m_dim-1;
138  /* k= m_dim -1 */
139  num=id/m_NN[k];
140  id=id%m_NN[k];
141  BDD bdd = m_bdd_interval[k].int_to_bdd(num);
142  for(k=m_dim-2; k >= 0; k--) {
143  num=id/m_NN[k];
144  id=id%m_NN[k];
145  bdd = bdd & m_bdd_interval[k].int_to_bdd(num);
146  }
147  return bdd;
148  }

183  std::vector<double> bdd_to_grid_points(const Cudd& manager, BDD bdd) {
184  if((!get_no_bdd_vars()) || bdd==manager.bddZero())
185  return {};
186  /* disable reordering (if enabled) */
187  Cudd_ReorderingType *method=nullptr;
188  if(manager.ReorderingStatus(method))
189  manager.AutodynDisable();
190  /* get variable ids */
191  auto var_id = get_bdd_var_ids();
192  /* find the variables in the support of the BDD but outside the SymbolicSet */
193  auto support_id = bdd.SupportIndices();
194  std::vector<BDD> out{};
195  for(const auto& id : support_id) {
196  if(std::find(std::begin(var_id), std::end(var_id), id)==std::end(var_id))
197  out.emplace_back(manager.bddVar(id));
198  }
199  /* remove those variables from the bdd */
200  if(out.size())
201  bdd = bdd.ExistAbstract(manager.computeCube(out));
202  /* limit the grid points in the grid */
203  for(const auto& interval : m_bdd_interval)
204  bdd = bdd & interval.get_all_elements();
205  /* init the vector of grid points to be returned */
206  abs_type no_gp = get_size(manager,bdd);
207  std::vector<double> gp(no_gp*m_dim);
208  for(abs_type i=0; i<no_gp; i++) {
209  for(int j=0; j<m_dim; j++)
210  gp[i*m_dim+j]=m_first[j];
211  }
212  /* set up iteration to iterate over BDD cubes */
213  DdManager* dd = manager.getManager();
214  int *cube;
215  CUDD_VALUE_TYPE value;
216  DdGen *gen;
217  abs_type counter=0;
218  /* iterate over BDD cubes */
219  Cudd_ForeachCube(dd,bdd.getNode(),gen,cube,value) {
220  abs_type offset=1;
221  for(int i=0; i<m_dim; i++) {
222  unsigned int no_vars = m_bdd_interval[i].get_no_bdd_vars();
223  for (unsigned int j=0; j<no_vars; j++) {
224  unsigned int id = m_bdd_interval[i].get_bdd_var_ids()[j];
225  if(cube[id]==1) {
226  for(abs_type k=0; k<offset; k++) {
227  gp[(counter+k)*m_dim+i]+=(abs_type{1}<<(no_vars-1-j))*m_eta[i];
228  }
229  }
230  /* take care of don't care */
231  if(cube[id]==2) {
232  for(abs_type k=0; k<offset; k++) {
233  for(int l=0; l<=i; l++) {
234  gp[(counter+k+offset)*m_dim+l]=gp[(counter+k)*m_dim+l];
235  }
236  }
237  for(abs_type k=0; k<offset; k++) {
238  gp[(counter+k+offset)*m_dim+i]+=(abs_type{1}<<(no_vars-1-j))*m_eta[i];
239  }
240  offset=(offset<<1);
241  }
242  }
243  }
244  counter+=offset;
245  }
246  /* reactivate reordering if it was enabled */
247  if(method!=nullptr)
248  manager.AutodynEnable(*method);
249  return gp;
250  }

293  std::vector<double> restriction(const Cudd& manager,
294  const BDD& bdd,
295  const grid_point_t& x,
296  std::vector<int> domain = {}) const {
297  /* compute indices in domain/codomain */
298  std::vector<int> codomain {};
299  /* fill in default values */
300  if(domain.size()==0) {
301  domain.resize(x.size());
302  std::iota(std::begin(domain),std::end(domain),0);
303  codomain.resize(m_dim-x.size());
304  std::iota(std::begin(codomain),std::end(codomain),x.size());
305  } else {
306  for(int i=0; i<m_dim; i++) {
307  if(std::find(std::begin(domain), std::end(domain), i) == std::end(domain))
308  codomain.push_back(i);
309  }
310  }
311  /* create SymbolicSet of the domain and codmain */
312  SymbolicSet set_dom(*this,domain);
313  SymbolicSet set_codom(*this,codomain);
314  /* extract bdd with restriction */
315  abs_type id = set_dom.xtoi(x);
316  BDD restricted = bdd & set_dom.id_to_bdd(id);
317  restricted = restricted.ExistAbstract(set_dom.get_cube(manager));
318  /* map restricted BDD to grid points */
319  return set_codom.bdd_to_grid_points(manager,restricted);
320  }
